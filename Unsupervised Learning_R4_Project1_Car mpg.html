cluster_range=range(2,10)
meanDistortions=[]
for k in cluster_range:
    model=KMeans(n_clusters=k)
    model.fit(df_cars)
    model.predict(df_cars)
    #meanDistortions.append(sum(np.min(cdist(df_cars, model.cluster_centers_, 'euclidean'), axis=1)) / df_cars.shape[0])
    meanDistortions.append(sum(np.min(cdist(df_cars, model.cluster_centers_, 'euclidean'), axis=1)))
    silhouette_avg=silhouette_score(df_cars,model.predict(df_cars),metric='minkowski')
    print ("At cluster {}, silhouette score is {}".format(k, silhouette_avg))
